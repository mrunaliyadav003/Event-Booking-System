db.createCollection("EventPublisher", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["publisherId", "title", "type", "date", "location", "cost", "maxNumberOfParticipants"],
      properties: {
        publisherId: { bsonType: "string" },
        title: { bsonType: "string" },
        type: { enum: ["sport", "concert", "other"] },
        date: { bsonType: "date" },
        location: { bsonType: "string" },
        cost: { bsonType: "number", minimum: 0 },
        maxNumberOfParticipants: { bsonType: "int", minimum: 1 }
      }
    }
  }
});

db.createCollection("Student", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["Name", "EmailId"],
      properties: {
        Name: { bsonType: "string" },
        EmailId: { bsonType: "string" },
	Subscribed: { bsonType: "bool" }


      }
    }
  }
});

db.createCollection("Rating", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["studentId", "eventId"],
      properties: {
        studentId: { bsonType: "string" },
        eventId: { bsonType: "string" },
        score: { bsonType: "number", minimum: 0 },
        comment: { bsonType: "string" },
        ratingDate:  { bsonType: "date" },
      }
    }
  }
});
